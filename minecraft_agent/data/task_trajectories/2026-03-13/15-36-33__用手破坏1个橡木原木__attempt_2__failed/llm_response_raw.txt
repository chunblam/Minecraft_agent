async function breakOneOakLog(bot) {
    // 1. 寻找附近的橡木原木
    const logs = bot.findNearbyBlocks("oak_log", 32, 10);
    if (logs.length === 0) {
        bot.chat("附近没有找到橡木原木，开始探索寻找。");
        // 随机选择一个方向探索
        const directions = ["north", "south", "east", "west"];
        const randomDir = directions[Math.floor(Math.random() * directions.length)];
        await exploreUntil(bot, randomDir, 64, () => {
            const foundLogs = bot.findNearbyBlocks("oak_log", 32, 10);
            return foundLogs.length > 0;
        });
    }
    // 重新获取原木位置
    const logsAfter = bot.findNearbyBlocks("oak_log", 32, 10);
    if (logsAfter.length === 0) {
        throw new Error("探索后仍未找到橡木原木");
    }
    const targetLog = logsAfter[0];
    bot.chat(`找到橡木原木在 ${targetLog.x}, ${targetLog.y}, ${targetLog.z}`);

    // 2. 移动到原木附近（距离2格以内）
    await moveToPosition(bot, targetLog.x, targetLog.y, targetLog.z, 2);

    // 3. 用手破坏1个橡木原木（空手）
    await equipItem(bot, "air", "hand");
    await mineBlock(bot, "oak_log", 1);

    // 4. 捡起掉落的物品
    await pickupNearbyItems(bot);
    bot.chat("成功破坏1个橡木原木并捡起掉落物。");
}
