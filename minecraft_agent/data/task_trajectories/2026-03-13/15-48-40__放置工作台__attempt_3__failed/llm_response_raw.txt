Explain: 上次错误是因为 `equipItem(bot, 'air', 'hand')` 试图装备"air"，但背包中没有"air"物品。实际上放置方块不需要装备任何物品，`placeItem` 控制原语会自动处理。应该直接调用 `placeItem` 而不需要先装备空手。

Plan:
1. 检查背包中是否有工作台（已有）。
2. 寻找合适的放置位置：在附近找固体方块（如草方块、泥土、石头），取其上方一格。
3. 直接调用 `placeItem` 放置工作台，跳过装备步骤。
4. 确保放置位置安全（不在脚下，不与实体重叠）。

Code:
```javascript
async function placeWorkbench(bot) {
    // 1. 检查背包中是否有工作台
    const workbench = bot.inventory.items().find(item => item.name === 'crafting_table');
    if (!workbench) {
        bot.chat("背包中没有工作台，需要先合成一个。");
        // 检查是否有足够的橡木木板来合成工作台
        const planks = bot.inventory.items().find(item => item.name === 'oak_planks');
        if (!planks || planks.count < 4) {
            bot.chat("橡木木板不足，需要先收集原木制作木板。");
            // 使用已注入的技能 collectMoreLogs 收集原木
            await collectMoreLogs(bot, { logType: 'oak_log', targetCount: 1 });
            // 合成木板
            await craftItem(bot, 'oak_planks', 4);
        }
        // 合成工作台
        await craftItem(bot, 'crafting_table', 1);
        bot.chat("已合成工作台。");
    } else {
        bot.chat(`背包中有 ${workbench.count} 个工作台。`);
    }

    // 2. 寻找合适的放置位置
    // 从当前位置附近找一个固体方块（例如草方块、泥土、石头）
    let solidBlocks = bot.findNearbyBlocks('grass_block', 32, 10)
        .concat(bot.findNearbyBlocks('dirt', 32, 10))
        .concat(bot.findNearbyBlocks('stone', 32, 10));
    if (solidBlocks.length === 0) {
        bot.chat("附近没有找到合适的固体方块，移动一下寻找。");
        // 随机方向探索寻找固体方块
        const directions = ['north', 'south', 'east', 'west'];
        const randomDir = directions[Math.floor(Math.random() * directions.length)];
        await exploreUntil(bot, randomDir, 64, () => {
            const found = bot.findNearbyBlocks('grass_block', 32, 10)
                .concat(bot.findNearbyBlocks('dirt', 32, 10))
                .concat(bot.findNearbyBlocks('stone', 32, 10));
            return found.length > 0;
        });
        // 重新获取固体方块
        solidBlocks = bot.findNearbyBlocks('grass_block', 32, 10)
            .concat(bot.findNearbyBlocks('dirt', 32, 10))
            .concat(bot.findNearbyBlocks('stone', 32, 10));
        if (solidBlocks.length === 0) {
            throw new Error("探索后仍未找到合适的固体方块放置工作台");
        }
    }
    
    // 选择第一个固体方块，取其上方一格作为放置位置
    const baseBlock = solidBlocks[0];
    const placePos = { x: baseBlock.x, y: baseBlock.y + 1, z: baseBlock.z };
    
    // 确保放置位置不在bot脚下（与当前位置水平距离至少2格）
    const botPos = bot.entity.position;
    const dx = placePos.x - botPos.x;
    const dz = placePos.z - botPos.z;
    const distSq = dx * dx + dz * dz;
    if (distSq < 4) { // 距离小于2格
        bot.chat("放置位置太近，寻找另一个位置。");
        // 找另一个固体方块
        for (const block of solidBlocks) {
            const candidatePos = { x: block.x, y: block.y + 1, z: block.z };
            const dx2 = candidatePos.x - botPos.x;
            const dz2 = candidatePos.z - botPos.z;
            if (dx2 * dx2 + dz2 * dz2 >= 4) {
                // 找到合适位置
                bot.chat(`将在位置 ${candidatePos.x}, ${candidatePos.y}, ${candidatePos.z} 放置工作台。`);
                await placeItem(bot, 'crafting_table', candidatePos);
                bot.chat("工作台放置完成！");
                return;
            }
        }
        // 如果所有候选位置都太近，先移动一下
        bot.chat("所有候选位置都太近，先移动一下。");
        await moveToPosition(bot, botPos.x + 3, botPos.y, botPos.z, 2);
        // 重新寻找固体方块
        solidBlocks = bot.findNearbyBlocks('grass_block', 32, 10)
            .concat(bot.findNearbyBlocks('dirt', 32, 10))
            .concat(bot.findNearbyBlocks('stone', 32, 10));
        if (solidBlocks.length === 0) {
            throw new Error("移动后仍未找到合适的固体方块");
        }
        const newBase = solidBlocks[0];
        placePos.x = newBase.x;
        placePos.y = newBase.y + 1;
        placePos.z = newBase.z;
    }
    
    bot.chat(`将在位置 ${placePos.x}, ${placePos.y}, ${placePos.z} 放置工作台。`);
    
    // 3. 直接放置工作台（placeItem会自动处理装备）
    await placeItem(bot, 'crafting_table', placePos);
    bot.chat("工作台放置完成！");
}
```
